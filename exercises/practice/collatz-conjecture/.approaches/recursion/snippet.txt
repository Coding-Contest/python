def count_steps(conjecture_steps, number):
    if number <= 1:
        return conjecture_steps
    return count_steps(
        conjecture_steps + 1, number * 3 + 1 if number % 2 else number / 2
    )

def steps(number):
    if number < 1:
        raise ValueError("Only positive integers are allowed")

    return count_steps(0, number)
